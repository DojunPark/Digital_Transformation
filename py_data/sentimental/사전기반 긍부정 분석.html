SentiWord = pd.read_json('./data/SentiWord_info.json',encoding='utf-8')

def neg_pos(word):
    temp = pd.DataFrame()
    global SentiWord
    temp = SentiWord[(SentiWord['word'] == word) | 
                     (SentiWord['word_root'] == word)]
    try:
        word_tul = (word,temp['polarity'][temp.index[0]])
    except IndexError:
        word_tul = (word,0)
    return word_tul
